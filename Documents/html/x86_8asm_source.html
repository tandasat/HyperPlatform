93 ; void __stdcall AsmVmmEntryPoint();
94 AsmVmmEntryPoint PROC
95  ; No need to save the flag registers since it is restored from the VMCS at
96  ; the time of vmresume.
97  pushad ; -4 * 8
98 
99  mov ecx, esp
100  push ecx
101  call VmmVmExitHandler@4 ; bool vm_continue = VmmVmExitHandler(guest_context);
102 
103  test al, al
104  jz exitVM ; if (!vm_continue) jmp exitVM
105 
106  popad
107  vmresume
108  jmp vmxError
109 
110 exitVM:
111  ; Executes vmxoff and ends virtualization
112  ; eax = Guest's eflags
113  ; edx = Guest's esp
114  ; ecx = Guest's eip for the next instruction
115  popad
116  vmxoff
117  jz vmxError ; if (ZF) jmp
118  jc vmxError ; if (CF) jmp
119  push eax
120  popfd ; eflags <= GurstFlags
121  mov esp, edx ; esp <= GuestRsp
122  push ecx
123  ret ; jmp AddressToReturn

132  int 3
